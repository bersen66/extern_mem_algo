set(SORT_ROOT ${CMAKE_CURRENT_SOURCE_DIR})
set(SRC_DIR ${SORT_ROOT}/src)

# ADDING SOURCES
list(APPEND ENGINE_SOURCES
    ${SRC_DIR}/sort.cpp
)
set(ENGINE_SOURCES ${ENGINE_SOURCES} PARENT_SCOPE)

#SETTING UP INCLUDE DIR
list(APPEND ENGINE_INCLUDE_DIRS 
    ${SORT_ROOT}/include
)

set(ENGINE_INCLUDE_DIRS ${ENGINE_INCLUDE_DIRS} PARENT_SCOPE)

# ADDING TESTS
if (BUILD_TESTS)
    set(TEST_DIR ${SORT_ROOT}/test)
    list(APPEND ENGINE_TESTS 
        ${TEST_DIR}/example.cpp
    )
    set(ENGINE_TESTS ${ENGINE_TESTS} PARENT_SCOPE)
endif()
