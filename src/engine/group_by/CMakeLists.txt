set(GROUP_BY_ROOT ${CMAKE_CURRENT_SOURCE_DIR})
set(SRC_DIR ${GROUP_BY_ROOT}/src)

# ADDING SOURCES
list(APPEND ENGINE_SOURCES
    ${SRC_DIR}/group_by.cpp
)

set(ENGINE_SOURCES ${ENGINE_SOURCES} PARENT_SCOPE)

#SETTING UP INCLUDE DIR
list(APPEND ENGINE_INCLUDE_DIRS 
    ${GROUP_BY_ROOT}/include
)

set(ENGINE_INCLUDE_DIRS ${ENGINE_INCLUDE_DIRS} PARENT_SCOPE)


# ADDING TESTS
if (BUILD_TESTS)
    set(TEST_DIR ${GROUP_BY_ROOT}/test)
    list(APPEND ENGINE_TESTS 
        ${TEST_DIR}/example.cpp
    )
    set(ENGINE_TESTS ${ENGINE_TESTS} PARENT_SCOPE)
endif()

# ADDING BENCHMARKS
if (BUILD_BENCHMARKS)
    set(BENCH_DIR ${GROUP_BY_ROOT}/benchmarks)
    list(APPEND ENGINE_BENCHMARKS
        ${BENCH_DIR}/example.cpp
    )
    set(ENGINE_BENCHMARKS ${ENGINE_BENCHMARKS} PARENT_SCOPE)
endif()
